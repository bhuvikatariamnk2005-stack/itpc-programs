#include <stdio.h>

// User defined Function for swapping two nubres(call by value)
//Function Declaration
void swapnumbers(int a, int b);

int main() {
    int a,b;
    printf("Enter two Numbers:\n");
    scanf("%d %d",&a,&b);

    printf("Before swapping: a = %d, b = %d\n", a, b);
    swapnumbers(a, b);
    printf("After swapping: a = %d, b = %d\n", a, b);

    return 0;
}

// Function definition 
void swapnumbers(int a, int b) {
    int temp;
    temp=a;
    a=b;
    b=temp; 
}
